/* CREATION TABLE joueur */

CREATE TABLE joueur
(
	numero NUMBER(4) PRIMARY KEY NOT NULL,
	nom VARCHAR2(50),
	prenom VARCHAR2(50),
	nationalite VARCHAR2(25),
	numeroEquipe NUMBER(4),
	nbSelecNat NUMBER(4),
	FOREIGN KEY (numeroEquipe) REFERENCES Equipe (numero)
);


/* CREATION TABLE equipe */

CREATE TABLE equipe
(
	numero NUMBER(4) PRIMARY KEY NOT NULL,
	nom VARCHAR2(50),
	capaciteStade NUMBER(10),
	budget NUMBER(10)
);

/* CREATION DE LA TABLE competition */

CREATE TABLE competition
(
	numero NUMBER(4) PRIMARY KEY NOT NULL,
	nom VARCHAR2(50),

/* CREATION DE LA TABLE participer */

CREATE TABLE participer
(
	numeroEquipe NUMBER(4),
	numeroCompet NUMBER(4),
	resultat NUMBER(6),
	FOREIGN KEY (numeroEquipe) REFERENCES equipe(numero),
	FOREIGN KEY (numeroCompet) REFERENCES competition(numero),
	PRIMARY KEY (numeroEquipe,numeroCompet)
);
