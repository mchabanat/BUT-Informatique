Agences:
Private Sub LECTURE_Click()
Dim db As DAO.Database
Dim db As DAO.Recordset
Set db = CurrentDb()
Set rs = db.OpenRecordset ("SELECT * FROM agence", dbOpenDynaset)
If Not rs.EOF Then
	MsgBox rs("Ville")
	rs.MoveNext 
End If
End Sub


Private Sub INSERTION_DblClick(Cancel As Integer)
Dim db As DAO.Database
Dim db As DAO.Recordset
Set db = CurrentDb()
Set tb_client = db.OpenRecordset ("AGENCE", dbOpenDynaset)
tb_client.AddNew
tb_client("CODE") = "AG007"
tb_client("VILLE") = "TOURS"
tb_client.Update
End Sub 


Private Sub MODIFICATION_MouseMove()
Dim db As DAO.Database
Dim db As DAO.Recordset
Set db = CurrentDb()
Set rs = db.OpenRecordset ("SELECT * FROM agence WHERE [ville] = 'TOURS' ", dbOpenDynaset)
If Not rs.EOF Then
	rs.edit
	rs("VILLE") = 'POITIERS'
	rs.Update
End If
End Sub


Private Sub SUPPRIMER_Click()
Dim db As DAO.Database
Dim db As DAO.Recordset
Set db = CurrentDb()
Set rs = db.OpenRecordset ("SELECT * FROM agence WHERE [ville] = 'POITIERS' ", dbOpenDynaset)
If Not rs.EOF Then
	rs.delete
	rs.MoveNext
End If
End Sub
